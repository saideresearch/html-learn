ফর্ম এলিমেন্ট

<form> এলিমেন্ট

form এলিমেন্ট ফর্মের মূল কন্টেইনার।

<form action="/submit" method="post">
  <!-- ফর্ম এলিমেন্ট এখানে বসবে -->
</form>

<label> এলিমেন্ট

label ট্যাগ ইনপুট ফিল্ডের জন্য লেবেল নির্ধারণ করে। for অ্যাট্রিবিউটে ইনপুটের id দিতে হয়।

<label for="username">ইউজারনেম:</label>
<input type="text" id="username" name="username">

<input> এলিমেন্ট

input ট্যাগ বিভিন্ন ধরনের ইনপুট নেওয়ার জন্য। type অ্যাট্রিবিউট দিয়ে ইনপুটের ধরণ নির্ধারণ করা হয়।

<input type="text" name="fullname" placeholder="আপনার নাম">
<input type="email" name="email" placeholder="ইমেইল">
<input type="password" name="password" placeholder="পাসওয়ার্ড">




<button> এলিমেন্ট

button ট্যাগ দিয়ে বাটন তৈরি করা হয়। type submit দিলে ফর্ম সাবমিট হয়।

<button type="submit">সাবমিট</button>
<button type="reset">রিসেট</button>
<button type="button" onclick="alert('ক্লিক')">ক্লিক করুন</button>

<fieldset> ও <legend>

fieldset দিয়ে সম্পর্কিত ফর্ম এলিমেন্ট গ্রুপ করা হয়। legend দিয়ে গ্রুপের শিরোনাম দেওয়া হয়।

<fieldset>
  <legend>ব্যক্তিগত তথ্য</legend>
  <label>নাম: <input type="text"></label><br>
  <label>ইমেইল: <input type="email"></label>
</fieldset>
ব্যক্তিগত তথ্য

<select> ও <option>

ড্রপডাউন মেনু তৈরি করতে।

<label for="country">দেশ:</label>
<select id="country" name="country">
  <option value="bd">বাংলাদেশ</option>
  <option value="in">ভারত</option>
  <option value="pk">পাকিস্তান</option>
</select>

<textarea>

একাধিক লাইনের টেক্সট ইনপুট নেওয়ার জন্য।

<label for="message">বার্তা:</label><br>
<textarea id="message" rows="4" cols="50">
আপনার বার্তা লিখুন...
</textarea>
