HTML5 APIs (Application Programming Interfaces)

পড়তে সময়: ৭০ মিনিট লেভেল: উন্নত

HTML5 APIs কী?

HTML5 APIs হলো জাভাস্ক্রিপ্ট-ভিত্তিক ইন্টারফেস যা ব্রাউজারের হার্ডওয়্যার, ফাইল সিস্টেম, নেটওয়ার্ক, স্টোরেজ এবং অন্যান্য ক্ষমতায় অ্যাক্সেস প্রদান করে। এই APIs ব্যবহার করে আপনি ওয়েব অ্যাপ্লিকেশনকে ডেস্কটপ অ্যাপের মতো শক্তিশালী এবং ইন্টারঅ্যাকটিভ করতে পারেন। জিওলোকেশন, ড্র্যাগ অ্যান্ড ড্রপ, অফলাইন স্টোরেজ, পুশ নোটিফিকেশন, ক্যামেরা অ্যাক্সেস, WebRTC ভিডিও কল—এই সবই HTML5 APIs-এর মাধ্যমে সম্ভব।

শিখবেন যা:

  • ক্লায়েন্ট-সাইড স্টোরেজ: localStorage, sessionStorage, IndexedDB
  • হার্ডওয়্যার অ্যাক্সেস: জিওলোকেশন, ক্যামেরা/মাইক্রোফোন (getUserMedia)
  • পারফরম্যান্স ও অবজারভার API: Intersection Observer, Resize Observer
  • ব্যাকগ্রাউন্ড প্রসেসিং: Web Workers, Service Workers
  • রিয়েল-টাইম যোগাযোগ: WebSockets, WebRTC, Server-Sent Events
  • ইউজার ইন্টারঅ্যাকশন: Drag & Drop, Clipboard API, Fullscreen API
  • নোটিফিকেশন ও পেজ কন্ট্রোল: Notification API, Page Visibility API, History API

জিওলোকেশন API (Geolocation API)

জিওলোকেশন API ব্যবহারকারীর বর্তমান অবস্থান (অক্ষাংশ, দ্রাঘিমাংশ) অ্যাক্সেস করতে দেয়। ইউজারের অনুমতি সাপেক্ষে এটি কাজ করে।

জিওলোকেশন উদাহরণ:

if ("geolocation" in navigator) {
  navigator.geolocation.getCurrentPosition(
    (position) => {
      const lat = position.coords.latitude;
      const lng = position.coords.longitude;
      console.log(`অবস্থান: ${lat}, ${lng}`);
    },
    (error) => {
      console.error("লোকেশন অ্যাক্সেস অস্বীকৃত:", error.message);
    }
  );
}

// অবস্থান ট্র্যাকিং
const watchId = navigator.geolocation.watchPosition(
  (position) => updateLocation(position),
  (error) => handleError(error)
);
// ট্র্যাকিং বন্ধ করতে: navigator.geolocation.clearWatch(watchId);

ব্যবহার: ম্যাপ সেবা, লোকেশন-ভিত্তিক সার্চ, ডেলিভারি ট্র্যাকিং, আবহাওয়া অ্যাপ ইত্যাদিতে জিওলোকেশন ব্যবহার করা হয়।

ড্র্যাগ অ্যান্ড ড্রপ API (Drag and Drop API)

ড্র্যাগ অ্যান্ড ড্রপ API ব্যবহারকারীকে HTML এলিমেন্ট ড্র্যাগ করে অন্য স্থানে ড্রপ করার সুযোগ দেয়। এটি ফাইল আপলোড, ক্যানভাস ডিজাইন, টাস্ক ম্যানেজমেন্ট অ্যাপে ব্যবহৃত হয়।

ড্র্যাগ অ্যান্ড ড্রপ উদাহরণ:

<div draggable="true" ondragstart="dragStart(event)">ড্র্যাগ করুন</div>
<div ondrop="drop(event)" ondragover="allowDrop(event)">এখানে ড্রপ করুন</div>

<script>
function dragStart(event) {
  event.dataTransfer.setData("text/plain", event.target.id);
}
function allowDrop(event) {
  event.preventDefault();
}
function drop(event) {
  event.preventDefault();
  const data = event.dataTransfer.getData("text/plain");
  const draggedElement = document.getElementById(data);
  event.target.appendChild(draggedElement);
}
</script>
📦 ড্র্যাগ করুন
📂 এখানে ড্রপ করুন
(ডেমো - সম্পূর্ণ ফাংশনালিটির জন্য JavaScript প্রয়োজন)

ওয়েব স্টোরেজ API (Web Storage API)

ওয়েব স্টোরেজ API ক্লায়েন্ট-সাইডে কী-ভ্যালু জোড়া সংরক্ষণের সুযোগ দেয়। এটি localStorage এবং sessionStorage নিয়ে গঠিত।

// localStorage - ডেটা স্থায়ী (ব্রাউজার বন্ধ হলেও থাকে)
localStorage.setItem('username', 'রহিম');
const user = localStorage.getItem('username');
localStorage.removeItem('username');
localStorage.clear();

// sessionStorage - ডেটা সেশন চলাকালীন (ট্যাব বন্ধ হলে মুছে যায়)
sessionStorage.setItem('tempData', 'সেশন ডাটা');
const temp = sessionStorage.getItem('tempData');

localStorage vs sessionStorage:

  • localStorage: ৫-১০MB সীমা, ডেটা স্থায়ী, সব ট্যাবে শেয়ার হয়
  • sessionStorage: ~৫MB সীমা, ট্যাব বন্ধে মুছে যায়, শুধু ওই ট্যাবে অ্যাক্সেসযোগ্য

ওয়েব ওয়ার্কার্স (Web Workers)

Web Workers ব্যাকগ্রাউন্ড থ্রেডে জাভাস্ক্রিপ্ট চালানোর সুযোগ দেয়। এটি UI থ্রেডকে ব্লক না করে ভারী গণনা, ডাটা প্রসেসিং, ফাইল পার্সিং করতে পারে।

// main.js
const worker = new Worker('worker.js');
worker.postMessage({ data: heavyData });
worker.onmessage = (event) => {
  console.log('রেজাল্ট:', event.data);
};

// worker.js
self.onmessage = (event) => {
  const result = heavyComputation(event.data);
  self.postMessage(result);
};

ব্যবহার: ইমেজ প্রসেসিং, বড় ডাটাসেট ফিল্টারিং, এনক্রিপশন, প্যারালাল প্রসেসিং—যেখানে UI ফ্রিজ এড়ানো প্রয়োজন।

getUserMedia (ক্যামেরা ও মাইক্রোফোন)

getUserMedia API ব্যবহারকারীর ক্যামেরা, মাইক্রোফোন এবং স্ক্রিন অ্যাক্সেস করতে দেয়। এটি ভিডিও কল, ফটো ক্যাপচার, ভয়েস রেকর্ডিং অ্যাপে ব্যবহৃত হয়।

navigator.mediaDevices.getUserMedia({ video: true, audio: true })
  .then((stream) => {
    const videoElement = document.querySelector('video');
    videoElement.srcObject = stream;
  })
  .catch((error) => {
    console.error('ক্যামেরা অ্যাক্সেস অস্বীকৃত:', error);
  });

// স্ক্রিন শেয়ারিং
navigator.mediaDevices.getDisplayMedia({ video: true })
  .then((stream) => { /* স্ক্রিন ক্যাপচার */ });
🎥 ক্যামেরা প্রিভিউ (অনুমতি প্রয়োজন)

নোটিফিকেশন API (Notification API)

নোটিফিকেশন API ডেস্কটপ পুশ নোটিফিকেশন দেখানোর সুযোগ দেয়। ইউজারের অনুমতি সাপেক্ষে এটি কাজ করে।

// অনুমতি চাওয়া
Notification.requestPermission().then(permission => {
  if (permission === 'granted') {
    new Notification('নতুন বার্তা!', {
      body: 'আপনার একটি নতুন মেসেজ পেয়েছেন',
      icon: 'icon.png',
      badge: 'badge.png'
    });
  }
});

// টাইমার বা ইভেন্টের সাথে সংযুক্ত
setTimeout(() => {
  new Notification('রিমাইন্ডার', { body: 'আপনার মিটিং শুরু হতে ৫ মিনিট বাকি' });
}, 60000);

ক্লিপবোর্ড API (Clipboard API)

ক্লিপবোর্ড API ব্যবহারকারীর ক্লিপবোর্ডে টেক্সট, ইমেজ, বা HTML কপি/পেস্ট করার সুযোগ দেয়। এটি নিরাপদ এবং আধুনিক ব্রাউজারে সাপোর্টেড।

// টেক্সট কপি
async function copyText(text) {
  try {
    await navigator.clipboard.writeText(text);
    console.log('কপি হয়েছে!');
  } catch (err) {
    console.error('কপি ব্যর্থ:', err);
  }
}

// ক্লিপবোর্ড থেকে পড়া
async function pasteText() {
  try {
    const text = await navigator.clipboard.readText();
    console.log('পেস্ট করা টেক্সট:', text);
  } catch (err) {
    console.error('পেস্ট ব্যর্থ:', err);
  }
}

// ইমেজ কপি
const imageBlob = await fetch('image.png').then(r => r.blob());
await navigator.clipboard.write([
  new ClipboardItem({ 'image/png': imageBlob })
]);

ইন্টারসেকশন অবজারভার API (Intersection Observer API)

Intersection Observer API কোনো এলিমেন্ট ভিউপোর্টে প্রবেশ করছে কি না তা পর্যবেক্ষণ করে। এটি লেজি লোডিং, ইনফিনিট স্ক্রল, অ্যানিমেশন ট্রিগার করার জন্য ব্যবহৃত হয়।

const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      // এলিমেন্ট দৃশ্যমান হয়েছে
      entry.target.classList.add('visible');
      // লেজি লোড ইমেজ
      const img = entry.target.querySelector('img');
      if (img && img.dataset.src) {
        img.src = img.dataset.src;
      }
    }
  });
}, { threshold: 0.5 }); // 50% দৃশ্যমান হলে ট্রিগার

// পর্যবেক্ষণ শুরু
const elements = document.querySelectorAll('.lazy-load');
elements.forEach(el => observer.observe(el));

ব্যবহার: ইমেজ লেজি লোডিং, ইনফিনিট স্ক্রল পেজিনেশন, অ্যানিমেশন ট্রিগার, অ্যাড ইম্প্রেশন ট্র্যাকিং।

ফুলস্ক্রিন API (Fullscreen API)

ফুলস্ক্রিন API যেকোনো HTML এলিমেন্টকে পুরো স্ক্রিনে দেখানোর সুযোগ দেয়। ভিডিও প্লেয়ার, গেম, প্রেজেন্টেশন অ্যাপে ব্যবহৃত হয়।

// ফুলস্ক্রিনে যান
function enterFullscreen(element) {
  if (element.requestFullscreen) {
    element.requestFullscreen();
  } else if (element.webkitRequestFullscreen) {
    element.webkitRequestFullscreen(); // Safari
  }
}

// ফুলস্ক্রিন থেকে বেরুন
function exitFullscreen() {
  if (document.exitFullscreen) {
    document.exitFullscreen();
  }
}

// ফুলস্ক্রিন পরিবর্তন মনিটর
document.addEventListener('fullscreenchange', () => {
  if (document.fullscreenElement) {
    console.log('ফুলস্ক্রিন মোডে');
  } else {
    console.log('স্বাভাবিক মোডে');
  }
});

পেজ ভিজিবিলিটি API (Page Visibility API)

পেজ ভিজিবিলিটি API ট্যাব সক্রিয় আছে কি না তা নির্ণয় করে। এটি ব্যাকগ্রাউন্ড ট্যাবে অ্যানিমেশন বন্ধ করা, ভিডিও পজ করা, বা রিসোর্স সেভ করার জন্য ব্যবহার করা হয়।

document.addEventListener('visibilitychange', () => {
  if (document.hidden) {
    console.log('ট্যাব লুকানো/ব্যাকগ্রাউন্ডে');
    // অ্যানিমেশন বন্ধ, ভিডিও পজ
    videoElement.pause();
  } else {
    console.log('ট্যাব দৃশ্যমান');
    // অ্যানিমেশন শুরু
    videoElement.play();
  }
});

// চেক করা
if (!document.hidden) {
  // ট্যাব দৃশ্যমান
}

হিস্ট্রি API (History API)

হিস্ট্রি API ব্রাউজারের ইতিহাস নিয়ন্ত্রণ করে। এটি SPA (Single Page Application)-তে URL পরিবর্তন করে পেজ রিলোড ছাড়াই নেভিগেশন সম্ভব করে।

// নতুন স্টেট পুশ করা (URL পরিবর্তন করে)
history.pushState({ page: 1 }, 'পেজ ১', '/page-1');

// স্টেট প্রতিস্থাপন (বর্তমান এন্ট্রি পরিবর্তন)
history.replaceState({ page: 2 }, 'পেজ ২', '/page-2');

// ব্যাক/ফরোয়ার্ড ইভেন্ট শোনা
window.addEventListener('popstate', (event) => {
  console.log('লোকেশন:', location.pathname);
  console.log('স্টেট:', event.state);
  // পেজ কন্টেন্ট আপডেট
  loadContent(location.pathname);
});

রিসাইজ অবজারভার API (Resize Observer API)

Resize Observer API এলিমেন্টের সাইজ পরিবর্তন পর্যবেক্ষণ করে। এটি responsive ডিজাইন, অ্যাডাপটিভ লেআউট, canvas রিসাইজে ব্যবহৃত হয়।

const resizeObserver = new ResizeObserver(entries => {
  for (let entry of entries) {
    const { width, height } = entry.contentRect;
    console.log(`এলিমেন্ট: ${entry.target} - সাইজ: ${width} x ${height}`);
    // কন্টেন্ট অ্যাডাপ্ট করুন
    if (width < 500) {
      entry.target.classList.add('mobile-view');
    } else {
      entry.target.classList.remove('mobile-view');
    }
  }
});

const element = document.querySelector('.resizable-container');
resizeObserver.observe(element);

সার্ভার-সেন্ট ইভেন্ট (Server-Sent Events - SSE)

SSE সার্ভার থেকে ক্লায়েন্টে একমুখী রিয়েল-টাইম ডাটা পাঠানোর পদ্ধতি। এটি WebSocket-এর চেয়ে সহজ এবং স্বয়ংক্রিয় রিকানেক্ট সাপোর্ট করে।

// ক্লায়েন্ট-সাইড
const eventSource = new EventSource('/events');

eventSource.onmessage = (event) => {
  console.log('নতুন ডাটা:', event.data);
  updateUI(JSON.parse(event.data));
};

eventSource.addEventListener('custom-event', (event) => {
  console.log('কাস্টম ইভেন্ট:', event.data);
});

// সংযোগ বন্ধ করতে
eventSource.close();

// সার্ভার-সাইড (Node.js উদাহরণ)
res.writeHead(200, {
  'Content-Type': 'text/event-stream',
  'Cache-Control': 'no-cache',
  'Connection': 'keep-alive'
});
res.write(`data: ${JSON.stringify({ message: 'হ্যালো' })}\n\n`);

ফাইল API (File API)

File API ব্যবহারকারীর লোকাল ফাইল পড়তে এবং প্রসেস করতে দেয়। ইমেজ প্রিভিউ, CSV পার্সিং, ড্র্যাগ-অ্যান্ড-ড্রপ আপলোডে ব্যবহৃত হয়।

<input type="file" id="fileInput" multiple>

document.getElementById('fileInput').addEventListener('change', (event) => {
  const files = event.target.files;
  for (let file of files) {
    console.log(`নাম: ${file.name}, সাইজ: ${file.size} bytes, টাইপ: ${file.type}`);
    
    // ইমেজ প্রিভিউ
    if (file.type.startsWith('image/')) {
      const reader = new FileReader();
      reader.onload = (e) => {
        const img = document.createElement('img');
        img.src = e.target.result;
        document.body.appendChild(img);
      };
      reader.readAsDataURL(file);
    }
    
    // টেক্সট ফাইল পড়া
    if (file.type === 'text/plain') {
      const reader = new FileReader();
      reader.onload = (e) => {
        console.log('ফাইল কন্টেন্ট:', e.target.result);
      };
      reader.readAsText(file);
    }
  }
});

WebRTC (Web Real-Time Communication)

WebRTC ব্রাউজারের মধ্যে পিয়ার-টু-পিয়ার অডিও, ভিডিও, ডাটা শেয়ারিং সুবিধা দেয়। ভিডিও কল, ফাইল শেয়ারিং, স্ক্রিন শেয়ারিং অ্যাপে ব্যবহৃত হয়।

// লোকাল স্ট্রিম
const localStream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });

// RTCPeerConnection তৈরি
const peerConnection = new RTCPeerConnection(configuration);

// ট্র্যাক যোগ করা
localStream.getTracks().forEach(track => {
  peerConnection.addTrack(track, localStream);
});

// ICE ক্যান্ডিডেট হ্যান্ডলিং
peerConnection.onicecandidate = (event) => {
  if (event.candidate) {
    // অন্য পিয়ারে পাঠান
    sendToPeer(event.candidate);
  }
};

// অফার তৈরি
const offer = await peerConnection.createOffer();
await peerConnection.setLocalDescription(offer);
// অফার অন্য পিয়ারে পাঠান
